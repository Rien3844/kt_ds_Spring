CREATE TABLE "FILES" (
"FILE_NUM" NUMBER(2, 0) NOT NULL,
"FILE_GROUP_ID" VARCHAR2 (18 CHAR) NOT NULL,
"OBFUSCATE_NAME" VARCHAR2 (50 CHAR) NOT NULL,
"DISPLAY_NAME" VARCHAR2 (250 CHAR) NOT NULL,
"EXTEND_NAME" VARCHAR2 (8 CHAR) NULL,
"FILE_LENGTH" NUMBER(36, 0) NOT NULL,
"FILE_PATH" VARCHAR2 (500 CHAR) NOT NULL
);

 

COMMENT ON COLUMN "FILES"."FILE_NUM" IS '그룹 별 순번';
COMMENT ON COLUMN "FILES"."EXTEND_NAME" IS '확장자';
COMMENT ON COLUMN "FILES"."FILE_LENGTH" IS '첨부파일 크기(바이트 기준)';
COMMENT ON COLUMN "FILES"."FILE_PATH" IS '파일 시스템상 경로';

 

ALTER TABLE "FILES" ADD CONSTRAINT "PK_FILES" PRIMARY KEY (
"FILE_NUM",
"FILE_GROUP_ID"
);
